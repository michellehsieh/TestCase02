OK * 實作一個清單頁面，顯示欄位有「客戶名稱、聯絡人數量、銀行帳戶數量」共三個欄位，此功能要在資料庫中建立一個檢視表，並且匯入 EDMX。
	CREATE VIEW [dbo].[客戶資料統計表]
	AS SELECT [dbo].[客戶資料].[ID], [dbo].[客戶資料].[客戶名稱], 
				COUNT([dbo].[客戶聯絡人].[ID]) AS 聯絡人數量, 
				COUNT([dbo].[客戶銀行資訊].[ID]) AS 帳戶數量
         FROM [dbo].[客戶資料] LEFT JOIN [dbo].[客戶聯絡人] 
					       ON [dbo].[客戶資料].[ID] = [dbo].[客戶聯絡人].[客戶ID]
				    LEFT JOIN [dbo].[客戶銀行資訊]
					       ON [dbo].[客戶資料].[ID] = [dbo].[客戶銀行資訊].[客戶ID]
	    WHERE [dbo].[客戶資料].[是否已刪除] = 0
     GROUP BY [dbo].[客戶資料].[ID], [dbo].[客戶資料].[客戶名稱]
